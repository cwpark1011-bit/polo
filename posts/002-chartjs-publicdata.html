2) CSV로 바꾸는 법

  2. 아래 스크립트의 fetch() 주석 해제

<script>

const values = [18,22,38,33,42,51];

const ctx = document.getElementById('chart').getContext('2d');
const chart = new Chart(ctx,{
  type:'line',
  data:{ labels, datasets:[{ label:'키워드 트렌드(예시)', data:values, tension:0.3 }] },
